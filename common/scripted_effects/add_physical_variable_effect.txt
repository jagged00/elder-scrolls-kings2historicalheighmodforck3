add_breast_variable_effect = {
	if = {
		limit = {
			exists = scope:real_father 
			exists = scope:mother
			trigger_if = {
				limit = { 
					exists = scope:real_father 
					exists = scope:mother
				}
				scope:real_father = { is_alive = yes }
				scope:mother = { is_alive = yes }
				trigger_if = {
					limit = {
						scope:real_father = { is_alive = yes }
						scope:mother = { is_alive = yes }
					}
					scope:real_father = { has_variable = breast_size_gene_1 }
					scope:real_father = { has_variable = breast_size_gene_2 }
					scope:mother = { has_variable = breast_size_gene_1 }
					scope:mother = { has_variable = breast_size_gene_2 }
				}
			}
		}
		random = {
			chance = 50
			set_variable = {
				name = breast_size_gene_1
				value = scope:real_father.var:breast_size_gene_1
			}
		}
		random = {
			chance = 50
			set_variable = {
				name = breast_size_gene_2
				value = scope:mother.var:breast_size_gene_1
			}
		}
		if = {
			limit = { Not = { has_variable = breast_size_gene_1 } }
			set_variable = {
				name = breast_size_gene_1
				value = scope:real_father.var:breast_size_gene_2
			}
		}
		if = {
			limit = { Not = { has_variable = breast_size_gene_2 } }
			set_variable = {
				name = breast_size_gene_2
				value = scope:mother.var:breast_size_gene_2
			}
		}
	}
	else_if = {
		limit = {
			exists = scope:father
			exists = scope:mother
			Not = { exists = scope:real_father }
			trigger_if = {
				limit = { 
					exists = scope:father
					exists = scope:mother
					Not = { exists = scope:real_father }
				}
				scope:father = { is_alive = yes }
				scope:mother = { is_alive = yes }
				trigger_if = {
					limit = {
						scope:father = { is_alive = yes }
						scope:mother = { is_alive = yes }
					}
					scope:father = { has_variable = breast_size_gene_1 }
					scope:father = { has_variable = breast_size_gene_2 }
					scope:mother = { has_variable = breast_size_gene_1 }
					scope:mother = { has_variable = breast_size_gene_2 }
				}
			}
		}
		random = {
			chance = 50
			set_variable = {
				name = breast_size_gene_1
				value = scope:father.var:breast_size_gene_1
			}
		}
		random = {
			chance = 50
			set_variable = {
				name = breast_size_gene_2
				value = scope:mother.var:breast_size_gene_1
			}
		}
		if = {
			limit = { Not = { has_variable = breast_size_gene_1 } }
			set_variable = {
				name = breast_size_gene_1
				value = scope:father.var:breast_size_gene_2
			}
		}
		if = {
			limit = { Not = { has_variable = breast_size_gene_2 } }
			set_variable = {
				name = breast_size_gene_2
				value = scope:mother.var:breast_size_gene_2
			}
		}
	}
	else_if = {
		limit = { 
			exists = scope:real_father 
			trigger_if = {
				limit = { exists = scope:real_father }
				scope:real_father = { is_alive = yes }
				trigger_if = {
					limit = { scope:real_father = { is_alive = yes } }
					scope:real_father = { has_variable = breast_size_gene_1 }
					scope:real_father = { has_variable = breast_size_gene_2 }
				}
			}
		}
		random = {
			chance = 50
			set_variable = {
				name = breast_size_gene_1
				value = scope:real_father.var:breast_size_gene_1
			}
		}
		if = {
			limit = { Not = { has_variable = breast_size_gene_1 } }
			set_variable = {
				name = breast_size_gene_1
				value = scope:real_father.var:breast_size_gene_2
			}
		}
		set_breast_size_gene_2_effect = yes
	}
	else_if = {
		limit = { 
			exists = scope:father 
			Not = { exists = scope:real_father }
			trigger_if = {
				limit = { 
					exists = scope:father
					Not = { exists = scope:real_father }
				}
				scope:father = { is_alive = yes }
				trigger_if = {
					limit = { scope:father = { is_alive = yes } }
					scope:father = { has_variable = breast_size_gene_1 }
					scope:father = { has_variable = breast_size_gene_2 }
				}
			}
		}
		random = {
			chance = 50
			set_variable = {
				name = breast_size_gene_1
				value = scope:father.var:breast_size_gene_1
			}
		}
		if = {
			limit = { Not = { has_variable = breast_size_gene_1 } }
			set_variable = {
				name = breast_size_gene_1
				value = scope:father.var:breast_size_gene_2
			}
		}
		set_breast_size_gene_2_effect = yes
	}
	else_if = {
		limit = { 
			exists = scope:mother
			trigger_if = {
				limit = { exists = scope:mother }
				scope:mother = { is_alive = yes }
				trigger_if = {
					limit = { scope:mother = { is_alive = yes } }

				}
			}
		}
		random = {
			chance = 50
			set_variable = {
				name = breast_size_gene_2
				value = scope:mother.var:breast_size_gene_1
			}
		}
		if = {
			limit = { Not = { has_variable = breast_size_gene_2 } }
			set_variable = {
				name = breast_size_gene_2
				value = scope:mother.var:breast_size_gene_2
			}
		}
		set_breast_size_gene_1_effect = yes
	}
	else = {
		set_breast_size_gene_1_effect = yes
		set_breast_size_gene_2_effect = yes
	}
	random = {
		chance = breast_size_mutation_chance
		change_variable = {
			name = breast_size_gene_1
			add = {
				value = {
					fixed_range = { 
						max = breast_size_mutation_max
						min = breast_size_mutation_min
					}
				}
			}
		}
	}
	random = {
		chance = breast_size_mutation_chance
		change_variable = {
			name = breast_size_gene_2
			add = {
				value = {
					fixed_range = { 
						max = breast_size_mutation_max
						min = breast_size_mutation_min
					}
				}
			}
		}
	}
	if = {
		limit = { var:breast_size_gene_1 < 0 }
		remove_variable = breast_size_gene_1
		set_variable = {
			name = breast_size_gene_1
			value = 0
		}
	}
	if = {
		limit = { var:breast_size_gene_2 < 0 }
		remove_variable = breast_size_gene_2
		set_variable = {
			name = breast_size_gene_2
			value = 0
		}
	}
	if = {
		limit = { check_breast_gene_max_min > 0 }
		set_variable = {
			name = breast_size_genetic
			value = {
				value = {
					fixed_range = { 
						max = var:breast_size_gene_1
						min = var:breast_size_gene_2
					}
				}
				add = {
					fixed_range = { 
						max = var:breast_size_gene_1
						min = var:breast_size_gene_2
					}
				}
				divide = 2
			}
		}
	}
	else = {
		set_variable = {
			name = breast_size_genetic
			value = {
				value = {
					fixed_range = {
						min = var:breast_size_gene_1
						max = var:breast_size_gene_2
					}
				}
				add = {
					fixed_range = {
						min = var:breast_size_gene_1
						max = var:breast_size_gene_2
					}
				}
				divide = 2
			}
		}
	}
	trigger_event = {
		on_action = on_carn_tits_trait
	}
}
