breast_size_gui = {
	scope = character

	is_shown = {
		trigger_if = {
			limit = { is_female = yes }
			Not = { has_character_flag = chest_gender_morph }
		}
		trigger_else = {
			has_character_flag = chest_gender_morph
		}
		is_alive = yes
		is_adult = yes
		trigger_if = {
			limit = {
				is_alive = yes
			}
			has_variable = breast_size_genetic
		}
	}
}
