massive_butt_value = 200

massive_butt_ratio = {
	if = {
		limit = { has_variable = butt_hip_size_total_variable }
		if = {
			limit = { var:butt_hip_size_total_variable >= massive_butt_value }
			value = var:butt_hip_size_total_variable
			divide = massive_butt_value
		}
		else_if = {
			limit = { var:butt_hip_size_total_variable >= big_butt_value }
			value = var:butt_hip_size_total_variable
			subtract = big_butt_value
			divide = {
				value = massive_butt_value
				subtract = big_butt_value
			}
		}
		else = {
			value = 0
		}
	}
	else = {
		value = 0
	}
}
