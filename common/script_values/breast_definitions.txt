massive_tits_value = 100

massive_tits_ratio = {
	if = {
		limit = { has_variable = actual_breast_size_variable }
		if = {
			limit = { var:actual_breast_size_variable >= massive_tits_value }
			value = var:actual_breast_size_variable
			divide = massive_tits_value
		}
		else_if = {
			limit = { var:actual_breast_size_variable >= big_tits_value }
			value = var:actual_breast_size_variable
			subtract = big_tits_value
			divide = {
				value = massive_tits_value
				subtract = big_tits_value
			}
		}
		else = {
			value = 0
		}
	}
	else = {
		value = 0
	}
}
