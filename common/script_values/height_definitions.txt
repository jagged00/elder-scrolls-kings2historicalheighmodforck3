giant_value = 100
tall_value = 70
average_height_value = 50

giant_ratio = {
	if = {
		limit = { 
			has_variable = actual_height_variable
			var:actual_height_variable >= 0
		}
		if = {
			limit = { var:actual_height_variable >= giant_value }
			value = var:actual_height_variable
			divide = giant_value
		}
		else_if = {
			limit = { var:actual_height_variable >= tall_value }
			value = var:actual_height_variable
			subtract = tall_value
			divide = {
				value = giant_value
				subtract = tall_value
			}
		}
		else = {
			value = 0
		}
	}
	else = {
		value = 0
	}
}
