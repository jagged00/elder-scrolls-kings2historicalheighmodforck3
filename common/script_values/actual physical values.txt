actual_butt_size = {
	if = {
		limit = { 
			trigger_if = {
				limit = { is_alive = yes }
				has_variable = butt_size_genetic
				has_variable = cbo_muscles
				has_variable = cbo_lower_body_muscles
			}
			trigger_else = {
				always = no
			}
		}
		value = var:butt_size_genetic
		if = {
			limit = { current_weight <= -25 }
			add = {
				value = current_weight
				add = 25
				multiply = 2
			}
			min = 0
		}
		if = {
			limit = { has_variable = cbo_lower_body_muscles }
			multiply = {
				value = 100
				subtract = {
					value = actual_cbo_lower_body_muscles
					max = 67
				}
				divide = 100
			}
			add = actual_cbo_lower_body_muscles
		}
		add = butt_size_from_upper_body_muscle
		if = {
			limit = { 
				has_variable = puberty_value
				is_adult = no
			}
			multiply = capped_puberty_value
		}
		else_if = {
			limit = { 
				is_adult = no
			}
			multiply = 0
		}
		if = {
			limit = { current_weight > 0 }
			add = {
				value = current_weight
				max = 50
				if = {
					limit = { has_variable = lower_body_full_metabolic_gene }
					multiply = 1.32
				}
				else_if = {
					limit = { has_variable = lower_body_half_metabolic_gene }
					multiply = 1
				}
				else_if = {
					limit = { has_variable = body_mid_metabolic_gene }
					multiply = 0.66
				}
				else_if = {
					limit = { has_variable = upper_body_half_metabolic_gene }
					multiply = 0.34
				}
			}
		}
	}
	else = {
		value = 0
	}
}

pregnant_stomach_size = {
	if = {
		limit = { is_pregnant = yes }
		value = 100
		multiply = {
			add = pregnancy_days
			subtract = 30
			divide = 220 # PREGNANCY_DAYS is 280; we want 1 month at 100%
			max = 1
		}
		if = {
			# Overweight females should have less visible pregnancy, so we change the 100% pregnant value
			limit = { upper_body_fat > 0 }
			multiply = {
				value = 100
				subtract = upper_body_fat # An absolute and normalized value derived from character body weight
				multiply = 0.01
				max = 1
				min = 0
			}
		}
		min = 0
	}
	else = {
		value = 0
	}
}
