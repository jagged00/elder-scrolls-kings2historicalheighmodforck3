bulky_arms_value = 100

bulky_arms_ratio = {
	if = {
		limit = { 
			has_variable = visual_arm_muscle_variable
			has_variable = upper_body_fat_variable
		}
		if = {
			limit = { var:visual_arm_muscle_variable >= bulky_arms_value }
			value = var:visual_arm_muscle_variable
			divide = bulky_arms_value
		}
		else_if = {
			limit = { var:visual_arm_muscle_variable >= muscular_arms_value }
			value = var:visual_arm_muscle_variable
			subtract = muscular_arms_value
			divide = {
				value = bulky_arms_value
				subtract = muscular_arms_value
			}
		}
		else = {
			value = 0
		}
	}
	else = {
		value = 0
	}
}
