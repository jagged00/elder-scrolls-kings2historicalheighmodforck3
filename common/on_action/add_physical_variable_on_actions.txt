on_physical_variable = {
	on_actions = { 
		on_breast_variable
		on_butt_variable
		on_hips_variable
		on_waist_variable
		on_shoulders_variable
		on_height_variable
		on_muscle_variable
		on_lactation_genetic
		on_add_visual_age_variable
		on_metabolic_variable
		on_add_height_growth_delay
		on_puberty_start_age
	}
}

on_breast_variable = {
	trigger = { 
		Or = {
			not = { has_variable = breast_size_genetic }
			not = { has_variable = breast_size_gene_1 }
			not = { has_variable = breast_size_gene_2 }
		}
	}
	effect = {
		add_breast_variable_effect = yes
	}
}
