		NOT = {
			is_target_in_global_variable_list = {
				name = triggered_bookmark_events
				target = flag:gene_variable_seeding
				}
			}
		}

		add_to_global_variable_list = {
			name = triggered_bookmark_events
			target = flag:gene_variable_seeding
		}

				trigger_event = { on_action = on_physical_variable }
		}
	}
