				template = pregnant
				value = {
					value = pregnant_stomach_size

				}
			}
			morph = {
				mode = modify
				gene = gene_areolas
				value = {
					value = var:lactation_genetic

						value = pregnancy_days
						max = 120

			morph = {
				mode = modify_multiply

				mode = modify
				gene = gene_bs_waist
				value = {
					value = pregnancy_days

					max = 1

			base = 0
			modifier = {

				trigger_if = {
					limit = { is_alive = yes }
					has_variable = lactation_genetic
					has_variable = post_pregnancy_breasts
